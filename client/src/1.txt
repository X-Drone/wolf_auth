src/
├── app/                    # Входная точка приложения, инициализация
│   ├── providers/          # Обёртки приложения (Router, Store, Theme)
│   └── App.tsx
├── shared/                 # Переиспользуемые элементы между фичами
│   ├── lib/                # Утилиты, хуки, вспомогательные функции
│   ├── ui/                 # Глобальные UI-компоненты (Button, Input, Modal)
│   ├── types/              # Общие TypeScript-типы (если используешь)
│   └── config/             # Глобальные настройки (API_URL, env)
├── entities/               # Бизнес-сущности (User, Friend, Task)
│   ├── User/               # Логика, связанная с пользователем
│   ├── Friend/             # Логика, связанная с друзьями
│   └── Task/               # Если интегрировано
├── features/               # Фичи (связанные с пользовательским сценарием)
│   ├── auth/               # Вход/регистрация
│   ├── profile/            # Редактирование профиля
│   ├── friend-list/        # Список друзей
│   └── add-friend/         # Добавление друга
├── widgets/                # Композиционные блоки (шапка, навигация)
│   ├── Header/
│   ├── Sidebar/
│   └── ProfileMenu/
├── pages/                  # Страницы (сборка фич и виджетов)
│   ├── LoginPage/
│   ├── ProfilePage/
│   └── FriendsPage/
├── widgets/                # Компоненты уровня виджета (если нужно вынести)
├── services/               # Взаимодействие с API (axios, fetch)
│   ├── api/
│   └── auth/
├── styles/                 # Глобальные стили и темы (CSS-модули, dark theme)
└── index.tsx
